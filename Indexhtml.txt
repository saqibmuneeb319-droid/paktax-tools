<script>
  // Tax calculator
  document.getElementById("calc-tax").addEventListener("click", function() {
    const salary = parseFloat(document.querySelectorAll("input")[1].value);
    let tax = 0;
    if (salary > 50000) {
      tax = salary * 0.05;
    }
    document.getElementById("tax-result").innerText = "Estimated Tax: Rs. " + tax.toFixed(0);
  });

  // Currency converter (demo rates)
  document.getElementById("convert").addEventListener("click", function() {
    const pkr = parseFloat(document.getElementById("pkr-amount").value);
    const currency = document.getElementById("currency").value;
    let rate = 1;
    if (currency === "USD") rate = 280;
    if (currency === "EUR") rate = 300;
    if (currency === "SAR") rate = 75;
    const result = (pkr / rate).toFixed(2);
    document.getElementById("currency-result").innerText = pkr + " PKR = " + result + " " + currency;
  });
</script>
